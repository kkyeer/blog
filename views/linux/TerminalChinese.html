    终端zsh中文乱码问题解决

    kkyeer 6/20/2022 Terminal

    # 终端zsh中文乱码问题解决

    vi ~/.zshrc
# 添加下面2行到文件最后
export LC_ALL=en_US.UTF-8
export LANG=en_US.UTF-8
# 然后执行
source ~/.zshrc
